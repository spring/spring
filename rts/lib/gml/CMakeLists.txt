

ADD_DEFINITIONS(-DUSE_GML)

# USE_GML requires this
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${NO_TLS_DIRECT_SEG_REFS}")

# USE_GML_SIM requires USE_GML
ADD_DEFINITIONS(-DUSE_GML_SIM)

if (USE_GML_DEBUG)
	ADD_DEFINITIONS(-DUSE_GML_DEBUG)
endif (USE_GML_DEBUG)

FIND_PACKAGE(SDL2 REQUIRED)
FIND_PACKAGE_STATIC(GLEW 1.5.1 REQUIRED)
INCLUDE_DIRECTORIES(${SDL2_INCLUDE_DIR} ${GLEW_INCLUDE_DIR})

SET(gmlSources
		"ThreadSafeContainers.cpp"
		"gml.cpp"
		"gml_base.cpp"
		"speedy-tls.cpp"
	)

ADD_LIBRARY(gml STATIC EXCLUDE_FROM_ALL ${gmlSources})
TARGET_LINK_LIBRARIES(gml ${SDL2_LIBRARY} ${OPENGL_gl_LIBRARY} ${OPENGL_glu_LIBRARY} ${GLEW_LIBRARIES})
