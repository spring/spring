		"OscOutboundPacketStream"
		"OscTypes"
	)

	SET_TARGET_PROPERTIES(oscpack PROPERTIES COMPILE_FLAGS -DOSC_HOST_BIG_ENDIAN)

	SET_TARGET_PROPERTIES(oscpack PROPERTIES COMPILE_FLAGS -DOSC_HOST_LITTLE_ENDIAN)
ENDIF (APPLE)
